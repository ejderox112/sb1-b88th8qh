================================================================================
KEY CONFIGURATION CHANGES
================================================================================

vercel.json BEFORE (v2 format - Edge runtime risk):
{
  "version": 2,
  "builds": [{
    "src": "package.json",
    "use": "@vercel/static-build",
    "config": { "distDir": "web-build" }
  }],
  "routes": [{ "src": "/(.*)", "dest": "/$1" }]
}

vercel.json AFTER (Static site - No Edge runtime):
{
  "buildCommand": "npm run build:web",
  "outputDirectory": "web-build",
  "framework": null,  // ← PREVENTS EDGE RUNTIME
  "rewrites": [{
    "source": "/(.*)",
    "destination": "/index.html"  // ← SPA ROUTING
  }],
  "headers": [{
    "source": "/(.*).(js|css|...)",
    "headers": [{
      "key": "Cache-Control",
      "value": "public, max-age=31536000, immutable"
    }]
  }]
}
